% Declara variáveis para guardar os valores iniciais da altura do cabeçalho de da separação do rodapé (distência entre o fundo do bloco de texto e a linha de base do rodapé).
\newlength{\AlturaDoCabecalho}
\setlength{\AlturaDoCabecalho}{\baselineskip}
\newlength{\SeparacaoDoRodape}
\setlength{\SeparacaoDoRodape}{\AlturaDoCabecalho}
\addtolength{\SeparacaoDoRodape}{\headsep}
% Define a altura do cabeçalho e do rodapé da página
\setheadfoot{\AlturaDoCabecalho}{\SeparacaoDoRodape}
% Declara variáveis para guardar os valores iniciais das margens interna e externa.
\newlength{\MargemInterna}
\setlength{\MargemInterna}{15mm}
\newlength{\MargemExterna}
\setlength{\MargemExterna}{10mm}
% Declara variáveis para guardar os valores iniciais das margens superior e inferior.
\newlength{\MargemSuperior}
\setlength{\MargemSuperior}{5mm}
\newlength{\MargemInferior}
\setlength{\MargemInferior}{5mm}
%% Descomente as duas linhas abaixo para incluir o cabeçalho no cálculo da margem superior
\addtolength{\MargemSuperior}{\headheight}
\addtolength{\MargemSuperior}{\headsep}
%% Descomente a linha abaixo para incluir o rodapé no cálculo da margem inferior
\addtolength{\MargemInferior}{\footskip}

\setlrmarginsandblock{\MargemInterna}{\MargemExterna}{*}
